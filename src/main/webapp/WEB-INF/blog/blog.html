springMVC ajax提交后不进入success function方法

1、ajax提交的时候加上async:false

2、完整ajax提交方式：

--------------------------------------------------------------------

$().ready(function(){
   $("#register").click(function(){
       var params = $("#form").serialize();
       $.ajax({  
             type:'post',  
             url:'register/submit',
             date:params,
             dateType:'json',
             async:false,
             success:function(data, textStatus){  
                 alert(data.resultCode);
                 alert(data.errorInfo);
             }
      });  
  })
 })

--------------------------------------------------------------------
